        function showPage(pageId) {

            // Views

            document.querySelectorAll('.page-view').forEach(view => {

                view.classList.remove('active');

            });

            document.getElementById(`page-${pageId}`).classList.add('active');


            // Nav States

            document.querySelectorAll('.nav-btn').forEach(btn => {

                if(btn.getAttribute('data-page') === pageId) {

                    btn.classList.add('text-amber-600');

                    btn.classList.remove('text-slate-600');

                } else {

                    btn.classList.remove('text-amber-600');

                    btn.classList.add('text-slate-600');

                }

            });


            // Scroll Top

            window.scrollTo({ top: 0, behavior: 'smooth' });


            // Close Mobile Menu

            document.getElementById('mobile-menu').classList.add('hidden');


            // Re-init charts

            if(pageId === 'dev') initDevChart();

        }

        function updateServicesTotal() {

            const base = 50;

            const hasConseil = document.getElementById('srv-conseil').checked;

            const hasCompta = document.getElementById('srv-compta').checked;

            const hasPaie = document.getElementById('srv-paie').checked;

            const nbSalaries = parseInt(document.getElementById('nb-salaries').value) || 0;


            document.getElementById('paie-input').classList.toggle('hidden', !hasPaie);


            let total = base;

            if(hasConseil) total += 9.99;

            if(hasCompta) total += 9.99;

            if(hasPaie) total += (nbSalaries * 0.99);


            document.getElementById('services-total-display').innerText = total.toFixed(2) + ' €';

        }
